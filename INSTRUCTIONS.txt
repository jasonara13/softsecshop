Git pull the latest
-------------------
cd /tmp/
git clone --single-branch --branch dev https://github.com/Alcaeusd/softsecshop.git
cd softsecshop

Database setup
--------------
mysql -u root -p
create database secshop;
\q


mysql -u root -p secshop < accounts.sql
mysql -u root -p secshop < products.sql
mysql -u root -p secshop < orders.sql

sudo mysql -u root -p

CREATE USER 'secshopuser'@'localhost' IDENTIFIED BY 'softsecpassword';
GRANT ALL ON secshop.* TO 'secshopuser'@'localhost' IDENTIFIED BY 'softsecpassword' WITH GRANT OPTION;

FLUSH PRIVILEGES;

systemctl restart mysql

Reminder: remember to make necessary changes at the db.ini according to your setup (username, password, database)

nginx setup
-----------
cd /tmp/softsecshop/
cp nginxconfs/nginx.conf /etc/nginx/nginx.conf
cp nginxconfs/softsecshop /etc/nginx/sites-available/
rm /etc/nginx/sites-enabled/softsecshop
ln -s /etc/nginx/sites-available/softsecshop /etc/nginx/sites-enabled/softsecshop

website setup
-------------
cd /tmp/softsecshop/
rm -rf /var/www/*
mkdir /var/www/softsecshop
cp -R website/* /var/www/softsecshop/
cp db.ini /var/www/
chown -R www-data:www-data /var/www/*
systemctl restart nginx
